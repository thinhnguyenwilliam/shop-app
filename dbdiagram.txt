Table users {
  id int [pk]  
  email varchar
  password varchar
  name varchar
  role int
  avatar varchar
  phone int
  created_at datetime
  updated_at datetime
}


Table categories {
  id int [pk]  
  name varchar
  image text
}


Table products {
  id int [pk]
  name varchar
  price int
  oldprice int
  image text
  description text
  specification text  
  buyturn int //Stores the count of how many times the product has been purchased
  quantity int
  brand_id int
  category_id int
  created_at datetime
  updated_at datetime
}
